\begin{frame}{Abstract}
\begin{itemize}
    \item Implemented \textbf{Meta-WebGuard}, a multi-layered hybrid framework combining Calibrated SVM, \textbf{CNN-BiLSTM}, and an XGBoost Meta-Learner.
    \item Integrated a \textbf{Titan-98 Global Mix} dataset, bridging detection gaps using real-world "unseen" samples and CAPEC-aligned injection data.
    \item Developed a \textbf{Hybrid CNN-BiLSTM} architecture to capture both local spatial features and long-range sequential patterns.
    \item Achieved exceptional results: \textbf{98.81\% Ensemble Accuracy} and a record \textbf{99.23\% Precision} on 100\% unseen data samples.
    \item Optimized for production using \textbf{cost-sensitive learning}, applying 2.5x weighting to malicious samples to minimize false negatives.
\end{itemize}
\end{frame}

\begin{frame}{Introduction}
\begin{itemize}
    \item \textbf{Evolving Threat Landscape:} Modern attacks use sophisticated obfuscation to bypass standard filters; our system uses \texttt{urllib} decoding to expose hidden payloads.
    \item \textbf{Limitations of Rules:} Static signatures fail against mutated phishing; we use \textbf{Character-level NLP} to treat URLs as sequences.
    \item \textbf{Our Solution:} A hybrid "Stacking" approach that aggregates intelligence from statistical, textual, and neural models.
    \item \textbf{Hybrid Strength:} Combines the linguistic precision of \textbf{TF-IDF SVM} with the temporal and spatial memory of \textbf{CNN-BiLSTM}.
    \item \textbf{Meta-Learning:} Employs \textbf{XGBoost} to arbitrate between foundation models, identifying the most reliable indicators for each URL.
\end{itemize}
\end{frame}

\begin{frame}{Problem Statement}
\begin{itemize}
    \item \textbf{Detection Gap:} Traditional firewalls struggle with brand-spoofing; our script identifies "Look-alike" domains using a \textbf{High-Trust Whitelist}.
    \item \textbf{Static vs. Dynamic:} Fixed systems are bypassed by entropy-rich payloads; we utilize \textbf{Shannon Entropy} metrics to flag randomized strings.
    \item \textbf{Class Imbalance:} Malicious URLs are often rarer in datasets; our implementation uses \textbf{Stratified Sampling} and sample weighting to ensure robust training.
    \item \textbf{Cross-Domain Fragility:} Models often overfit to specific sources; we merged \textbf{Cisco Umbrella}, \textbf{PhishTank}, and \textbf{CAPEC} to ensure global generalization.
\end{itemize}
\end{frame}

\begin{frame}{Objectives}
\begin{itemize}
    \item Build a hybrid pipeline integrating \textbf{LinearSVC, CNN-BiLSTM, and XGBoost} for 4-class URL classification.
    \item Achieve \textbf{$>$98\% Ensemble Accuracy} to validate the system's \textbf{Zero-Day readiness}.
    \item Utilize \textbf{Calibrated Probability Scoring} to provide high-precision detection (Proven: \textbf{99.23\% Precision}).
    \item Develop a \textbf{Functional Web Interface} for real-time URL testing and visualization of detection results.
    
\end{itemize}
\end{frame}

% Phase 1: Massive Multi-Source Data Acquisition
\begin{frame}{Phases of Project (1/7)}
\textbf{Phase 1: Titan-98 Global Dataset Consolidation} \\[0.2cm]
Integrated multiple high-diversity sources to ensure model generalization:
\begin{itemize}
    \item \textbf{Benign Pool:} Cisco Umbrella and Majestic Million top domains.
    \item \textbf{Malicious Pool:} PhishTank and Active Phishing databases.
    \item \textbf{Payload Injection:} CAPEC-aligned dataset (Injection \& Manipulation).
    \item \textbf{The "Unseen" Bridge:} Specialized dataset used to bridge the 98\% accuracy gap through diverse URL structures.
    \item \textbf{Data Cleaning:} Protocol stripping, URL decoding, and stratified sampling (Normal, Phishing, Injection, Manipulation).
\end{itemize}

\end{frame}

% Phase 2: Feature Engineering & Domain Intelligence
\begin{frame}{Phases of Project (2/7)}
\textbf{Phase 2: Advanced Feature Engineering} \\[0.2cm]
Three distinct feature extraction pipelines were developed:
\begin{itemize}
    \item \textbf{Textual Vectorization:} TF-IDF with character n-grams (1--3) and 25,000 max features.
    \item \textbf{Sequential Encoding:} Character-level tokenization with padding (Max length: 550) for neural processing.
    \item \textbf{Domain Heuristics:} Manual extraction of Shannon Entropy, digit density, and special character counts.
    \item \textbf{Trust-Aware Logic:} Comparison against a `HIGH\_TRUST\_DOMAINS` whitelist to detect Brand Spoofing (e.g., "paypa1-support.com").
\end{itemize}
\end{frame}

% Phase 3: Calibrated SVM Foundation
\begin{frame}{Phases of Project (3/7)}
\textbf{Phase 3: Calibrated Linear SVC Development} \\[0.2cm]
Implementation of the first ensemble member:
\begin{itemize}
    \item \textbf{Model:} Linear Support Vector Classification (LinearSVC).
    \item \textbf{Calibration:} Wrapped in `CalibratedClassifierCV` to provide probability estimates instead of raw scores.
    \item \textbf{Optimization:} C-parameter tuning and squared hinge loss for high-dimensional TF-IDF space.
    \item \textbf{Validation:} Achieved high baseline accuracy on textual signatures.
\end{itemize}


\end{frame}

% Phase 4: Bi-Directional Deep Learning
\begin{frame}{Phases of Project (4/7)}
\textbf{Phase 4: Hybrid CNN-BiLSTM Neural Architecture} \\[0.2cm]
Deep learning model focused on multi-scale feature extraction:
\begin{itemize}
    \item \textbf{Spatial Layer:} \textbf{Conv1D layer} captures local character motifs (sliding windows) to detect attack fragments.
    \item \textbf{Temporal Processing:} Two stacked \textbf{Bidirectional LSTM} layers to capture long-range contextual intent.
    \item \textbf{Regularization:} SpatialDropout1D and standard Dropout (0.2--0.3) to prevent overfitting.
    \item \textbf{Softmax Output:} Multi-class classification (98.37\% internal accuracy) for distinct threat categories.
\end{itemize}

\end{frame}

% Phase 5: Meta-Learner & Ensemble Stacking
\begin{frame}{Phases of Project (5/7)}
\textbf{Phase 5: XGBoost Stacking Meta-Learner} \\[0.2cm]
The core innovation: A "Model of Models" architecture.
\begin{itemize}
    \item \textbf{Input Layer:} Stacked probabilities from SVM + CNN + Scaled Hand-crafted features.
    \item \textbf{The Meta-Learner:} \textbf{XGBoost Classifier} (300 estimators, max depth 8).
    \item \textbf{GPU Acceleration:} Utilized `hist` tree method with CUDA for rapid ensemble training.
    \item \textbf{Detection Bias:} Applied 2.5x sample weighting to malicious entries to prioritize lowering False Negatives.
\end{itemize}


\end{frame}

% Phase 6: Model Hardening & Evaluation
\begin{frame}{Phases of Project (6/7)}
\textbf{Phase 6: Evaluation & Precision Tuning} \\[0.2cm]
Rigorous testing against the "Unseen" data samples:
\begin{itemize}
    \item \textbf{Metrics:} Evaluated via Stratified Cross-Validation on the Meta-Learner layer.
    \item \textbf{Trust Analysis:} Verified the effectiveness of the `HIGH\_TRUST\_DOMAINS` logic in reducing False Positives for major brands.
    \item \textbf{Complexity Analysis:} Balanced the computational cost of BiLSTMs with the speed of XGBoost.
\end{itemize}


\end{frame}

% Phase 7: Serialization & Deployment Ready
\begin{frame}{Phases of Project (7/7)}
\textbf{Phase 7: System Serialization & Portability} \\[0.2cm]
Finalizing the pipeline for real-world integration:
\begin{itemize}
    \item \textbf{Artifact Persistence:} Saving 7 core components: Vectorizer, SVM, Keras Hybrid Model, Tokenizer, Scaler, Meta-Learner, and Label Encoder.
    \item \textbf{Pipeline Integrity:} Ensuring the preprocessing logic in `strip\_protocol` and `extract\_features` is identical between training and inference.
    \item \textbf{Optimization:} Models saved in `.pkl` and `.keras` formats for minimum loading latency.
\end{itemize}

\end{frame}

\begin{frame}{Algorithm: Global Data Consolidation (1/5)}
\begin{algorithm}[H]
\small
\begin{algorithmic}[1]

    \State \textbf{Step 1: Multi-Source Integration}
    \State \quad Load Benign (Cisco/Majestic) and PhishTank datasets
    \State \quad Import CAPEC-aligned Injection/Manipulation data
    \State \quad Inject "New Unseen" dataset to bridge accuracy gaps
    \State \quad Normalize labels: Normal, Phishing, Injection, Manipulation
    
    \State \textbf{Step 2: URL Normalization \& Cleaning}
    \State \quad Strip protocols (http/https) and 'www.' prefixes
    \State \quad Perform URL decoding (unquote) to expose payloads
    \State \quad Remove duplicates and perform stratified sampling
    \State \quad Final counts: 120k Benign, 80k Phishing, + Attack classes

\end{algorithmic}
\end{algorithm}
\end{frame}

\begin{frame}{Algorithm: Feature Engineering Architecture (2/5)}
\begin{algorithm}[H]
\small
\begin{algorithmic}

  \State \textbf{Step 3: Multi-Modal Feature Extraction}
  \State \hspace{1.2em}(a) \textbf{NLP:} Fit TfidfVectorizer (Max: 25k, N-grams: 1--3)
  \State \hspace{1.2em}(b) \textbf{Neural:} Tokenize URLs (char-level) and pad to length 550
  \State \hspace{1.2em}(c) \textbf{Structural:} Extract Shannon Entropy and digit density
  \State \hspace{1.2em}(d) \textbf{Security Logic:} Detect Brand Spoofing via Trust List
  \State \hspace{1.2em}\quad (i) Identify target brands (Paypal, Apple, Microsoft, etc.)
  \State \hspace{1.2em}\quad (ii) Flag if brand in domain BUT not in \textit{HIGH\_TRUST\_DOMAINS}

\end{algorithmic}
\end{algorithm}
\end{frame}

\begin{frame}{Algorithm: Calibrated SVM \& Deep Learning (3/5)}
\begin{algorithm}[H]
\small
\begin{algorithmic}

  \State \textbf{Step 4: Train Foundation Model A (Calibrated SVM)}
  \State \hspace{1.2em}(a) Initialize LinearSVC with $C=1.0$ and dual=False
  \State \hspace{1.2em}(b) Wrap in CalibratedClassifierCV (3-fold) for probabilities
  \State \hspace{1.2em}(c) Fit on TF-IDF matrix to learn textual signatures

  \State \textbf{Step 5: Train Foundation Model B (Hybrid CNN-BiLSTM)}
  \State \hspace{1.2em}(a) Layer 1: 64-dim Character Embedding
  \State \hspace{1.2em}(b) Layer 2: \textbf{Conv1D (64 filters, kernel=3)} for spatial features
  \State \hspace{1.2em}(c) Layer 3-4: Stacked Bidirectional LSTM (64 units each)
  \State \hspace{1.2em}(d) Output: Softmax activation for model-level probability distribution

\end{algorithmic}
\end{algorithm}
\end{frame}

\begin{frame}{Algorithm: Meta-Learner Stacking Ensemble (4/5)}
\begin{algorithm}[H]
\small
\begin{algorithmic}

  \State \textbf{Step 6: Feature Fusion \& Weighting}
  \State \hspace{1.2em}(a) Generate $P(class|SVM)$ and $P(class|CNN)$ probabilities
  \State \hspace{1.2em}(b) Concatenate: $[P_{svm}, P_{cnn}, HandFeats_{scaled}]$
  \State \hspace{1.2em}(c) Define Sample Weights: 2.5x penalty for Malicious URLs

  \State \textbf{Step 7: Train XGBoost Meta-Learner}
  \State \hspace{1.2em}(a) Config: 300 estimators, depth 8, learning rate 0.05
  \State \hspace{1.2em}(b) Enable GPU-accelerated Histogram tree method
  \State \hspace{1.2em}(c) Train XGBoost on fused features with cost-sensitivity

\end{algorithmic}
\end{algorithm}
\end{frame}

\begin{frame}{Algorithm: Inference \& Persistence (5/5)}
\begin{algorithm}[H]
\small
\begin{algorithmic}

  \State \textbf{Step 8: Model Serialization}
  \State \hspace{1.2em}(a) Save 7-key pipeline components (pkl/keras formats)
  \State \hspace{1.2em}(b) Persist LabelEncoder for categorical mapping

  \Procedure{ClassifyURL}{NewUrl}
    \State 1. Extract Hand-crafted features + Brand Spoofing flag

  \EndProcedure

\end{algorithm}
\end{frame}

\begin{frame}{Implementation (1/5) - Titan-98 Data Engineering}
\section{Implementation Details}
\subsection{Massive Hybrid Data Sourcing}
\begin{itemize}
    \item \textbf{Sources:} Integrated \textbf{Cisco Umbrella}, \textbf{Majestic Million}, \textbf{PhishTank}, and \textbf{CAPEC} datasets for a global threat profile.
    \item \textbf{Global Mix:} Incorporated a specialized "Unseen" dataset to bridge the 98\% accuracy gap and improve generalization.
    \item \textbf{Preprocessing:} Implemented protocol stripping, URL decoding, and character-level tokenization with a 10,000-word vocabulary.
\end{itemize}

\subsection{Feature Engineering}
\begin{itemize}
    \item \textbf{NLP Vectorization:} TF-IDF with character n-grams (1--3) and 25,000 max features to capture syntax.
    \item \textbf{Sequence Prep:} Standardized URL input to \texttt{MAX\_LEN = 550} for sequential deep learning.
    \item \textbf{Manual Heuristics:} Extraction of Shannon Entropy, digit density, and Brand-Spoofing flags.
\end{itemize}
\end{frame}

\begin{frame}{Implementation (2/5) - Foundation Model: Calibrated SVM}
\subsection{Calibrated SVM Development}
\begin{itemize}
    \item \textbf{Model:} \texttt{CalibratedClassifierCV(LinearSVC(dual=False, C=1.0))} using a 3-fold cross-validation.
    \item \textbf{Purpose:} Generates calibrated probability distributions ($P(class|URL)$) instead of hard binary labels.
    \item \textbf{Training:} Fitted on a sparse TF-IDF matrix to identify high-frequency malicious keywords.
\end{itemize}

\subsection{Role in Hybrid Framework}
\begin{itemize}
    \item \textbf{Signature Detection:} Provides a rapid textual baseline for the ensemble.
    \item \textbf{Input Provider:} Passes class probabilities as high-quality features to the final Meta-Learner.
\end{itemize}
\end{frame}

\begin{frame}{Implementation (3/5) - Deep Bi-Directional Architecture}
\subsection{Sequential & Spatial Feature Extraction}
\begin{itemize}
    \item \textbf{Conv1D Layer:} Extracts local character motifs (3-gram sliding windows) to identify attack signatures.
    \item \textbf{Stacked Bi-LSTM:} Two layers of \textbf{Bidirectional LSTMs} (64 units each) to capture long-range URL dependencies.
    \item \textbf{Regularization:} Integrated \textbf{SpatialDropout1D} and standard Dropout (0.2--0.3) for model robustness.
\end{itemize}

\subsection{Training Configuration}
\begin{itemize}
    \item \textbf{Performance:} Reached \textbf{98.37\% Validation Accuracy} over 15 epochs.
    \item \textbf{Optimization:} Adam optimizer with cost-sensitive sample weighting (2.5x).
\end{itemize}
\end{frame}

\begin{frame}{Implementation (4/5) - XGBoost Stacking Ensemble}
\subsection{The Meta-Learning Layer}
\begin{itemize}
    \item \textbf{Architecture:} A \textbf{Stacked Ensemble} using \textbf{XGBoost} as the primary decision arbitrator.
    \item \textbf{Feature Fusion:} Concatenates SVM probabilities, Bi-LSTM softmax outputs, and scaled hand-crafted features.
    \item \textbf{Parameters:} 300 estimators, max depth of 8, and a 0.05 learning rate for granular boundary detection.
\end{itemize}

\subsection{Cost-Sensitive Training}
\begin{itemize}
    \item \textbf{Weighted Penalties:} Applied a \textbf{2.5x importance weight} to malicious URLs to minimize False Negatives.
    \item \textbf{Hardware Acceleration:} Utilized the \texttt{hist} tree method with \textbf{CUDA} for high-speed ensemble training.
\end{itemize}
\end{frame}

\begin{frame}{Implementation (5/5) - Validation \& Persistence}
\subsection{The Hybrid Decision Pipeline}
\begin{itemize}
    \item \textbf{Meta-Ensemble:} Final classification driven by XGBoost arbitrator.
    \item \textbf{Breakthrough Metrics:} Achieved **99.23\% Precision** and **98.81\% Ensemble Accuracy**.
\end{itemize}

\subsection{System Integration}
\begin{itemize}
    \item \textbf{Artifact Persistence:} All 7 pipeline components (Tokenizer, Scaler, SVM, Keras Hybrid, XGBoost, etc.) saved via \texttt{joblib}.
    \item \textbf{Deployment Ready:} Modular design allows for real-time classification through a unified feature extraction script.
\end{itemize}
